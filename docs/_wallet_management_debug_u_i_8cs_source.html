172 async void UpdateMainAssetTokenAmount()
173 {
174 var balance = await web3.Eth.GetBalance.SendRequestAsync(GetAccountAddress());
175 var balanceInEther = Web3.Convert.FromWei(balance.Value);
176
177 var data = (int)account.ChainId switch
178 {
179 (int)SupportedChain.Polygon => new TokenFetchData("MATIC", "Matic", ENSService.ENS_ZERO_ADDRESS, balanceInEther),
180 (int)SupportedChain.Mumbai => new TokenFetchData("MATIC", "Matic", ENSService.ENS_ZERO_ADDRESS, balanceInEther),
181 _ => new TokenFetchData("ETH", "Ethereum", ENSService.ENS_ZERO_ADDRESS, balanceInEther),
182 };
183
184 tokenAssetsUI.UpdateTokenData(data);
185
186 var chainWrappedETHAddress = walletSDKConfig.GetChainConfig((int)account.ChainId).WrappedETHAddress;
187 if (!string.IsNullOrEmpty(chainWrappedETHAddress))
188 {
189 var wrappedETHSymbol = await FetchTokenSymbol(chainWrappedETHAddress);
190 var wrappedETHName = await FetchTokenName(chainWrappedETHAddress);
191 var wrappedETHBalance = await GetTokenBalance(chainWrappedETHAddress, GetAccountAddress());
192 var wrappedETHData = new TokenFetchData(wrappedETHSymbol, wrappedETHName, chainWrappedETHAddress, wrappedETHBalance);
193 tokenAssetsUI.UpdateTokenData(wrappedETHData);
194 }
195 }

204 async Task<string> FetchTokenSymbol(string address)
205 {
206 try
207 {
208 if (!address.StartsWith("0x") || address.Length != 42)
209 {
210 Debug.LogError("Invalid Ethereum address format.");
211 return string.Empty;
212 }
213
214 var symbolHandler = web3.Eth.GetContractQueryHandler<SymbolFunction>();
215 return await symbolHandler.QueryAsync<string>(address);
216 }
217 catch (RpcResponseException rpcEx)
218 {
219 Debug.LogError("RPC Response Error: " + rpcEx.Message);
220 return string.Empty;
221 }
222 catch (Exception e)
223 {
224 Debug.LogException(e);
225 return string.Empty;
226 }
