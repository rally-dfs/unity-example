31/*
32 * 32-bit integer manipulation macros (little endian)
33 */
34#ifndef GET_UINT32_LE
35#define GET_UINT32_LE(n,b,i) \
36{ \
37 (n) = ( (uint32_t) (b)[(i) ] ) \
38 | ( (uint32_t) (b)[(i) + 1] << 8 ) \
39 | ( (uint32_t) (b)[(i) + 2] << 16 ) \
40 | ( (uint32_t) (b)[(i) + 3] << 24 ); \
41}
42#endif
43
44#ifndef PUT_UINT32_LE
45#define PUT_UINT32_LE(n,b,i) \
46{ \
47 (b)[(i) ] = (uint8_t) ( ( (n) ) & 0xFF ); \
48 (b)[(i) + 1] = (uint8_t) ( ( (n) >> 8 ) & 0xFF ); \
49 (b)[(i) + 2] = (uint8_t) ( ( (n) >> 16 ) & 0xFF ); \
50 (b)[(i) + 3] = (uint8_t) ( ( (n) >> 24 ) & 0xFF ); \
51}
52#endif

251/*
252 * RIPEMD-160 process buffer
253 */
254void ripemd160_Update( RIPEMD160_CTX *ctx, const uint8_t *input, uint32_t ilen )
255{
256 uint32_t fill;
257 uint32_t left;
258
259 if( ilen == 0 )
260 return;
261
262 left = ctx->total[0] & 0x3F;
263 fill = 64 - left;
264
265 ctx->total[0] += (uint32_t) ilen;
266 ctx->total[0] &= 0xFFFFFFFF;
267
268 if( ctx->total[0] < (uint32_t) ilen )
269 ctx->total[1]++;
270
271 if( left && ilen >= fill )
272 {
273 memcpy( (void *) (ctx->buffer + left), input, fill );
274 ripemd160_process( ctx, ctx->buffer );
275 input += fill;
276 ilen -= fill;
277 left = 0;
278 }
279
280 while( ilen >= 64 )
281 {
282 ripemd160_process( ctx, input );
283 input += 64;
284 ilen -= 64;
285 }
286
287 if( ilen > 0 )
288 {
289 memcpy( (void *) (ctx->buffer + left), input, ilen );
290 }
291}

301/*
302 * RIPEMD-160 final digest
303 */
304void ripemd160_Final( RIPEMD160_CTX *ctx, uint8_t output[20] )
305{
306 uint32_t last, padn;
307 uint32_t high, low;
308 uint8_t msglen[8];
309
310 high = ( ctx->total[0] >> 29 )
311 | ( ctx->total[1] << 3 );
312 low = ( ctx->total[0] << 3 );
313
314 PUT_UINT32_LE( low, msglen, 0 );
315 PUT_UINT32_LE( high, msglen, 4 );
316
317 last = ctx->total[0] & 0x3F;
318 padn = ( last < 56 ) ? ( 56 - last ) : ( 120 - last );
319
320 ripemd160_Update( ctx, ripemd160_padding, padn );
321 ripemd160_Update( ctx, msglen, 8 );
322
323 PUT_UINT32_LE( ctx->state[0], output, 0 );
324 PUT_UINT32_LE( ctx->state[1], output, 4 );
325 PUT_UINT32_LE( ctx->state[2], output, 8 );
326 PUT_UINT32_LE( ctx->state[3], output, 12 );
327 PUT_UINT32_LE( ctx->state[4], output, 16 );
328}
