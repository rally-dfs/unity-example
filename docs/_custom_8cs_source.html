134 private byte[] HashStruct(IDictionary<string, MemberDescription[]> types, string primaryType, IEnumerable<MemberValue> message)
135 {
136 using (var memoryStream = new MemoryStream())
137 using (var writer = new BinaryWriter(memoryStream))
138 {
139 var encodedType = EncodeType(types, primaryType);
140 var typeHash = Sha3Keccack.Current.CalculateHash(Encoding.UTF8.GetBytes(encodedType));
141 writer.Write(typeHash);
142
143 EncodeData(writer, types, message);
144
145 writer.Flush();
146 return Sha3Keccack.Current.CalculateHash(memoryStream.ToArray());
147 }

150 private static string EncodeType(IDictionary<string, MemberDescription[]> types, string typeName)
151 {
152 var encodedTypes = EncodeTypes(types, typeName);
153 var encodedPrimaryType = encodedTypes.Single(x => x.Key == typeName);
154 var encodedReferenceTypes = encodedTypes.Where(x => x.Key != typeName).OrderBy(x => x.Key).Select(x => x.Value);
155 var fullyEncodedType = encodedPrimaryType.Value + string.Join(string.Empty, encodedReferenceTypes.ToArray());
156
157 return fullyEncodedType;
158 }

160 private static IList<KeyValuePair<string, string>> EncodeTypes(IDictionary<string, MemberDescription[]> types, string currentTypeName)
161 {
162 var currentTypeMembers = types[currentTypeName];
163 var currentTypeMembersEncoded = currentTypeMembers.Select(x => x.Type + " " + x.Name);
164 var result = new List<KeyValuePair<string, string>>
165 {
166 new KeyValuePair<string, string>(currentTypeName, currentTypeName + "(" + string.Join(",", currentTypeMembersEncoded.ToArray()) + ")")
167 };
168
169 result.AddRange(currentTypeMembers.Select(x => ConvertToElementType(x.Type)).Distinct().Where(IsReferenceType).SelectMany(x => EncodeTypes(types, x)));
170
171 return result;

183 internal static bool IsReferenceType(string typeName)
184 {
185 switch (typeName)
186 {
187 // TODO: specify more precise conditions
188 case var bytes when new Regex("bytes\\d+").IsMatch(bytes):
189 case var @uint when new Regex("uint\\d+").IsMatch(@uint):
190 case var @int when new Regex("int\\d+").IsMatch(@int):
191 case "bytes":
192 case "string":
193 case "bool":
194 case "address":
195 return false;
196 case var array when array.Contains("["):
197 return false;
198 default:
199 return true;
200 }

203 private void EncodeData(BinaryWriter writer, IDictionary<string, MemberDescription[]> types, IEnumerable<MemberValue> memberValues)
204 {
205 foreach (var memberValue in memberValues)
206 {
207 switch (memberValue.TypeName)
208 {
209 case var refType when IsReferenceType(refType):
210 {
211 writer.Write(HashStruct(types, memberValue.TypeName, (IEnumerable<MemberValue>)memberValue.Value));
212 break;
213 }
214 case "string":
215 {
216 var value = Encoding.UTF8.GetBytes((string)memberValue.Value);
217 var abiValueEncoded = Sha3Keccack.Current.CalculateHash(value);
218 writer.Write(abiValueEncoded);
219 break;
220 }
221 case "bytes":
222 {
223 byte[] value;
224 if (memberValue.Value is string)
225 {
226 value = ((string)memberValue.Value).HexToByteArray();
227 }
228 else
229 {
230 value = (byte[])memberValue.Value;
231 }
232 var abiValueEncoded = Sha3Keccack.Current.CalculateHash(value);
233 writer.Write(abiValueEncoded);
234 break;
235 }
236 default:
237 {
238 if (memberValue.TypeName.Contains("["))
239 {
240 var items = (IList)memberValue.Value;
241 var itemsMemberValues = new List<MemberValue>();
242 foreach (var item in items)
243 {
244 itemsMemberValues.Add(new MemberValue()
245 {
246 TypeName = memberValue.TypeName.Substring(0, memberValue.TypeName.LastIndexOf("[")),
247 Value = item
248 });
249 }
250 using (var memoryStream = new MemoryStream())
251 using (var writerItem = new BinaryWriter(memoryStream))
252 {
253 EncodeData(writerItem, types, itemsMemberValues);
254 writerItem.Flush();
255 writer.Write(Sha3Keccack.Current.CalculateHash(memoryStream.ToArray()));
256 }
257
258 }
259 else if (memberValue.TypeName.StartsWith("int") || memberValue.TypeName.StartsWith("uint"))
260 {
261 object value;
262 if (memberValue.Value is string)
263 {
264 BigInteger parsedOutput;
265 if (BigInteger.TryParse((string)memberValue.Value, out parsedOutput))
266 {
267 value = parsedOutput;
268 }
269 else
270 {
271 value = memberValue.Value;
272 }
273 }
274 else
275 {
276 value = memberValue.Value;
277 }
278 var abiValue = new ABIValue(memberValue.TypeName, value);
279 var abiValueEncoded = _abiEncode.GetABIEncoded(abiValue);
280 writer.Write(abiValueEncoded);
281 }
282 else
283 {
284 var abiValue = new ABIValue(memberValue.TypeName, memberValue.Value);
285 var abiValueEncoded = _abiEncode.GetABIEncoded(abiValue);
286 writer.Write(abiValueEncoded);
287 }
288 break;
289 }
290 }
291 }

299 public TypedData<TDomain> GenerateTypedData<T, TDomain>(T data, TDomain domain, string primaryTypeName)
300 {
301 var parameters = _parametersEncoder.GetParameterAttributeValues(typeof(T), data).OrderBy(x => x.ParameterAttribute.Order);
302
303 var typeMembers = new List<MemberDescription>();
304 var typeValues = new List<MemberValue>();
305 foreach (var parameterAttributeValue in parameters)
306 {
307 typeMembers.Add(new MemberDescription
308 {
309 Type = parameterAttributeValue.ParameterAttribute.Type,
310 Name = parameterAttributeValue.ParameterAttribute.Name
311 });
312
313 typeValues.Add(new MemberValue
314 {
315 TypeName = parameterAttributeValue.ParameterAttribute.Type,
316 Value = parameterAttributeValue.Value
317 });
318 }
319
320 var result = new TypedData<TDomain>
321 {
322 PrimaryType = primaryTypeName,
323 Types = new Dictionary<string, MemberDescription[]>
324 {
325 [primaryTypeName] = typeMembers.ToArray(),
326 ["EIP712Domain"] = MemberDescriptionFactory.GetTypesMemberDescription(typeof(TDomain))["EIP712Domain"]
327 },
328 Message = typeValues.ToArray(),
329 Domain = domain
330 };
331
332 return result;
333 }

509 {
510 var convertor = new ExpandoObjectConverter();
511 var jsonDeserialised = JsonConvert.DeserializeObject<IDictionary<string, object>>(json, convertor);
512 var types = jsonDeserialised["types"] as IDictionary<string, object>;
513 var typeMemberDescriptions = GetMemberDescriptions(types);
514 if (!typeMemberDescriptions.ContainsKey("EIP712Domain"))
515 {
516 var domainMemberDescription = MemberDescriptionFactory.GetTypesMemberDescription(typeof(DomainType)).FirstOrDefault();
517 typeMemberDescriptions.Add(domainMemberDescription.Key, domainMemberDescription.Value);
518 }
519
520 var domainValues = GetMemberValues((IDictionary<string, object>)jsonDeserialised["domain"], "EIP712Domain", typeMemberDescriptions);
521 var primaryType = string.Empty;
522
523 if (jsonDeserialised.ContainsKey("primaryType"))
524 {
525 primaryType = jsonDeserialised["primaryType"].ToString();
526 }
527 else
528 {
529 if (types.Count == 1)
530 {
531 primaryType = types.First().Key;
532 }
533 else
534 {
535 throw new Exception("Primary type not set");
536 }
537 }

539 var message = jsonDeserialised[messageKeySelector];
540 var messageValues = GetMemberValues((IDictionary<string, object>)message, primaryType, typeMemberDescriptions);

542 var rawTypedData = new TypedDataRaw()

544 DomainRawValues = domainValues.ToArray(),
