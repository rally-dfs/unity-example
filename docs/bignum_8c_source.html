34/* The structure bignum256 is an array of nine 32-bit values, which
35 * are digits in base 2^30 representation. I.e. the number
36 * bignum256 a;
37 * represents the value
38 * sum_{i=0}^8 a.val[i] * 2^{30 i}.
39 *
40 * The number is *normalized* iff every digit is < 2^30.
41 *
42 * As the name suggests, a bignum256 is intended to represent a 256
43 * bit number, but it can represent 270 bits. Numbers are usually
44 * reduced using a prime, either the group order or the field prime.
45 * The reduction is often partly done by bn_fast_mod, and similarly
46 * implicitly in bn_multiply. A *partly reduced number* is a
47 * normalized number between 0 (inclusive) and 2*prime (exclusive).
48 *
49 * A partly reduced number can be fully reduced by calling bn_mod.
50 * Only a fully reduced number is guaranteed to fit in 256 bit.
51 *
52 * All functions assume that the prime in question is slightly smaller
53 * than 2^256. In particular it must be between 2^256-2^224 and
54 * 2^256 and it must be a prime number.
55 */
56

59 return (((uint32_t)data[0]) << 24) |
60 (((uint32_t)data[1]) << 16) |

62 (((uint32_t)data[3]));
63}

89// convert a raw bigendian 256 bit value into a normalized bignum.
90// out_number is partly reduced (since it fits in 256 bit).
91void bn_read_be(const uint8_t *in_number, bignum256 *out_number)
92{
93 int i;
94 uint32_t temp = 0;
95 for (i = 0; i < 8; i++) {
96 // invariant: temp = (in_number % 2^(32i)) >> 30i
97 // get next limb = (in_number % 2^(32(i+1))) >> 32i
98 uint32_t limb = read_be(in_number + (7 - i) * 4);
99 // temp = (in_number % 2^(32(i+1))) << 30i
100 temp |= limb << (2*i);
101 // store 30 bits into val[i]
102 out_number->val[i]= temp & 0x3FFFFFFF;
103 // prepare temp for next round
104 temp = limb >> (30 - 2*i);
105 }
106 out_number->val[8] = temp;
107}

109// convert a normalized bignum to a raw bigendian 256 bit number.
110// in_number must be fully reduced.
111void bn_write_be(const bignum256 *in_number, uint8_t *out_number)
112{
113 int i;
114 uint32_t temp = in_number->val[8] << 16;
115 for (i = 0; i < 8; i++) {
116 // invariant: temp = (in_number >> 30*(8-i)) << (16 + 2i)
117 uint32_t limb = in_number->val[7 - i];
118 temp |= limb >> (14 - 2*i);
119 write_be(out_number + i * 4, temp);
120 temp = limb << (18 + 2*i);
121 }
122}

187{
188 int i;

234// Check whether a < b
235// a and b must be normalized
236// function is constant time (on some architectures, in particular ARM).
237int bn_is_less(const bignum256 *a, const bignum256 *b)
238{
239 int i;
240 uint32_t res1 = 0;
241 uint32_t res2 = 0;
242 for (i = 8; i >= 0; i--) {
243 res1 = (res1 << 1) | (a->val[i] < b->val[i]);
244 res2 = (res2 << 1) | (a->val[i] > b->val[i]);
245 }
246 return res1 > res2;
247}

261// Assigns res = cond ? truecase : falsecase
262// assumes that cond is either 0 or 1.
263// function is constant time.
264void bn_cmov(bignum256 *res, int cond, const bignum256 *truecase, const bignum256 *falsecase)
265{
266 int i;
267 uint32_t tmask = (uint32_t) -cond;
268 uint32_t fmask = ~tmask;
269
270 assert (cond == 1 || cond == 0);
271 for (i = 0; i < 9; i++) {
272 res->val[i] = (truecase->val[i] & tmask) |
273 (falsecase->val[i] & fmask);

277// shift number to the left, i.e multiply it by 2.
278// a must be normalized. The result is normalized but not reduced.
279void bn_lshift(bignum256 *a)
280{
281 int i;
282 for (i = 8; i > 0; i--) {
283 a->val[i] = ((a->val[i] << 1) & 0x3FFFFFFF) | ((a->val[i - 1] & 0x20000000) >> 29);
284 }
285 a->val[0] = (a->val[0] << 1) & 0x3FFFFFFF;
286}

288// shift number to the right, i.e divide by 2 while rounding down.
289// a must be normalized. The result is normalized.
290void bn_rshift(bignum256 *a)

294 a->val[i] = (a->val[i] >> 1) | ((a->val[i + 1] & 1) << 29);

299// multiply x by 1/2 modulo prime.
300// it computes x = (x & 1) ? (x + prime) >> 1 : x >> 1.
301// assumes x is normalized.
302// if x was partly reduced, it is also partly reduced on exit.
303// function is constant time.
304void bn_mult_half(bignum256 * x, const bignum256 *prime)
305{
306 int j;
307 uint32_t xodd = -(x->val[0] & 1);
308 // compute x = x/2 mod prime
309 // if x is odd compute (x+prime)/2
310 uint32_t tmp1 = (x->val[0] + (prime->val[0] & xodd)) >> 1;
311 for (j = 0; j < 8; j++) {
312 uint32_t tmp2 = (x->val[j+1] + (prime->val[j+1] & xodd));
313 tmp1 += (tmp2 & 1) << 29;
314 x->val[j] = tmp1 & 0x3fffffff;
315 tmp1 >>= 30;
316 tmp1 += tmp2 >> 1;
317 }
318 x->val[8] = tmp1;
319}

333// compute x = x mod prime by computing x >= prime ? x - prime : x.
334// assumes x partly reduced, guarantees x fully reduced.
335void bn_mod(bignum256 *x, const bignum256 *prime)
336{
337 const int flag = bn_is_less(x, prime); // x < prime
338 bignum256 temp;
339 bn_subtract(x, prime, &temp); // temp = x - prime
340 bn_cmov(x, flag, x, &temp);
341}

343// auxiliary function for multiplication.
344// compute k * x as a 540 bit number in base 2^30 (normalized).
345// assumes that k and x are normalized.
346void bn_multiply_long(const bignum256 *k, const bignum256 *x, uint32_t res[18])
347{
348 int i, j;
349 uint64_t temp = 0;
350
351 // compute lower half of long multiplication
352 for (i = 0; i < 9; i++)
353 {
354 for (j = 0; j <= i; j++) {
355 // no overflow, since 9*2^60 < 2^64
356 temp += k->val[j] * (uint64_t)x->val[i - j];
357 }
358 res[i] = temp & 0x3FFFFFFFu;
359 temp >>= 30;
360 }
361 // compute upper half
362 for (; i < 17; i++)
363 {
364 for (j = i - 8; j < 9 ; j++) {
365 // no overflow, since 9*2^60 < 2^64
366 temp += k->val[j] * (uint64_t)x->val[i - j];
367 }
368 res[i] = temp & 0x3FFFFFFFu;
369 temp >>= 30;
370 }
371 res[17] = temp & 0x3FFFFFFFu;
372}

374// auxiliary function for multiplication.
375// reduces res modulo prime.
376// assumes res normalized, res < 2^(30(i-7)) * 2 * prime
377// guarantees res normalized, res < 2^(30(i-8)) * 2 * prime
378void bn_multiply_reduce_step(uint32_t res[18], const bignum256 *prime, uint32_t i) {
379 // let k = i-8.
380 // on entry:
381 // 0 <= res < 2^(30k + 31) * prime
382 // estimate coef = (res / prime / 2^30k)
383 // by coef = res / 2^(30k + 256) rounded down
384 // 0 <= coef < 2^31
385 // subtract (coef * 2^(30k) * prime) from res
386 // note that we unrolled the first iteration
387 uint32_t j;
388 uint32_t coef = (res[i] >> 16) + (res[i + 1] << 14);
389 uint64_t temp = 0x2000000000000000ull + res[i - 8] - prime->val[0] * (uint64_t)coef;
390 assert (coef < 0x80000000u);
391 res[i - 8] = temp & 0x3FFFFFFF;
392 for (j = 1; j < 9; j++) {
393 temp >>= 30;
394 // Note: coeff * prime->val[j] <= (2^31-1) * (2^30-1)
395 // Hence, this addition will not underflow.
396 temp += 0x1FFFFFFF80000000ull + res[i - 8 + j] - prime->val[j] * (uint64_t)coef;
397 res[i - 8 + j] = temp & 0x3FFFFFFF;
398 // 0 <= temp < 2^61 + 2^30
399 }
400 temp >>= 30;
401 temp += 0x1FFFFFFF80000000ull + res[i - 8 + j];
402 res[i - 8 + j] = temp & 0x3FFFFFFF;
403 // we rely on the fact that prime > 2^256 - 2^224
404 // res = oldres - coef*2^(30k) * prime;
405 // and
406 // coef * 2^(30k + 256) <= oldres < (coef+1) * 2^(30k + 256)
407 // Hence, 0 <= res < 2^30k (2^256 + coef * (2^256 - prime))
408 // < 2^30k (2^256 + 2^31 * 2^224)
409 // < 2^30k (2 * prime)
410}

413// auxiliary function for multiplication.
414// reduces x = res modulo prime.
415// assumes res normalized, res < 2^270 * 2 * prime
416// guarantees x partly reduced, i.e., x < 2 * prime
417void bn_multiply_reduce(bignum256 *x, uint32_t res[18], const bignum256 *prime)
418{
419 int i;
420 // res = k * x is a normalized number (every limb < 2^30)
421 // 0 <= res < 2^270 * 2 * prime.
422 for (i = 16; i >= 8; i--) {
423 bn_multiply_reduce_step(res, prime, i);
424 assert(res[i + 1] == 0);
425 }
426 // store the result
427 for (i = 0; i < 9; i++) {
428 x->val[i] = res[i];
429 }
430}

432// Compute x := k * x (mod prime)
433// both inputs must be smaller than 180 * prime.
434// result is partly reduced (0 <= x < 2 * prime)
435// This only works for primes between 2^256-2^224 and 2^256.
436void bn_multiply(const bignum256 *k, bignum256 *x, const bignum256 *prime)
437{
438 uint32_t res[18] = {0};
439 bn_multiply_long(k, x, res);
440 bn_multiply_reduce(x, res, prime);
441 MEMSET_BZERO(res, sizeof(res));
442}

444// partly reduce x modulo prime
445// input x does not have to be normalized.
446// x can be any number that fits.
447// prime must be between (2^256 - 2^224) and 2^256
448// result is partly reduced, smaller than 2*prime
449void bn_fast_mod(bignum256 *x, const bignum256 *prime)
450{
451 int j;
452 uint32_t coef;
453 uint64_t temp;
454
455 coef = x->val[8] >> 16;
456 // substract (coef * prime) from x
457 // note that we unrolled the first iteration
458 temp = 0x2000000000000000ull + x->val[0] - prime->val[0] * (uint64_t)coef;
459 x->val[0] = temp & 0x3FFFFFFF;
460 for (j = 1; j < 9; j++) {
461 temp >>= 30;
462 temp += 0x1FFFFFFF80000000ull + x->val[j] - prime->val[j] * (uint64_t)coef;
463 x->val[j] = temp & 0x3FFFFFFF;
464 }
465}

467// square root of x = x^((p+1)/4)
468// http://en.wikipedia.org/wiki/Quadratic_residue#Prime_or_prime_power_modulus
469// assumes x is normalized but not necessarily reduced.
470// guarantees x is reduced
471void bn_sqrt(bignum256 *x, const bignum256 *prime)
472{
473 // this method compute x^1/2 = x^(prime+1)/4
474 uint32_t i, j, limb;
475 bignum256 res, p;
476 bn_one(&res);
477 // compute p = (prime+1)/4
478 memcpy(&p, prime, sizeof(bignum256));
479 bn_addi(&p, 1);
480 bn_rshift(&p);
481 bn_rshift(&p);
482 for (i = 0; i < 9; i++) {
483 // invariants:
484 // x = old(x)^(2^(i*30))
485 // res = old(x)^(p % 2^(i*30))
486 // get the i-th limb of prime - 2
487 limb = p.val[i];
488 for (j = 0; j < 30; j++) {
489 // invariants:
490 // x = old(x)^(2^(i*30+j))
491 // res = old(x)^(p % 2^(i*30+j))
492 // limb = (p % 2^(i*30+30)) / 2^(i*30+j)
493 if (i == 8 && limb == 0) break;
494 if (limb & 1) {
495 bn_multiply(x, &res, prime);
496 }
497 limb >>= 1;
498 bn_multiply(x, x, prime);
499 }
500 }
501 bn_mod(&res, prime);
502 memcpy(x, &res, sizeof(bignum256));
503 MEMSET_BZERO(&res, sizeof(res));
504 MEMSET_BZERO(&p, sizeof(p));
505}

509// in field G_prime, small but slow
510void bn_inverse(bignum256 *x, const bignum256 *prime)
511{
512 // this method compute x^-1 = x^(prime-2)
513 uint32_t i, j, limb;
514 bignum256 res;
515 bn_one(&res);
516 for (i = 0; i < 9; i++) {
517 // invariants:
518 // x = old(x)^(2^(i*30))
519 // res = old(x)^((prime-2) % 2^(i*30))
520 // get the i-th limb of prime - 2
521 limb = prime->val[i];
522 // this is not enough in general but fine for secp256k1 & nist256p1 because prime->val[0] > 1
523 if (i == 0) limb -= 2;
524 for (j = 0; j < 30; j++) {
525 // invariants:
526 // x = old(x)^(2^(i*30+j))
527 // res = old(x)^((prime-2) % 2^(i*30+j))
528 // limb = ((prime-2) % 2^(i*30+30)) / 2^(i*30+j)
529 // early abort when only zero bits follow
530 if (i == 8 && limb == 0) break;
531 if (limb & 1) {
532 bn_multiply(x, &res, prime);
533 }
534 limb >>= 1;
535 bn_multiply(x, x, prime);
536 }
537 }
538 bn_mod(&res, prime);
539 memcpy(x, &res, sizeof(bignum256));
540}

544// in field G_prime, big and complicated but fast
545// the input must not be 0 mod prime.
546// the result is smaller than prime
547void bn_inverse(bignum256 *x, const bignum256 *prime)
548{
549 int i, j, k, cmp;
550 struct combo {
551 uint32_t a[9];
552 int len1;
553 } us, vr, *odd, *even;
554 uint32_t pp[8];
555 uint32_t temp32;
556 uint64_t temp;
557
558 // The algorithm is based on Schroeppel et. al. "Almost Modular Inverse"
559 // algorithm. We keep four values u,v,r,s in the combo registers
560 // us and vr. us stores u in the first len1 limbs (little endian)
561 // and s in the last 9-len1 limbs (big endian). vr stores v and r.
562 // This is because both u*s and v*r are guaranteed to fit in 8 limbs, so
563 // their components are guaranteed to fit in 9. During the algorithm,
564 // the length of u and v shrinks while r and s grow.
565 // u,v,r,s correspond to F,G,B,C in Schroeppel's algorithm.
566
567 // reduce x modulo prime. This is necessary as it has to fit in 8 limbs.
568 bn_fast_mod(x, prime);
569 bn_mod(x, prime);
570 // convert x and prime to 8x32 bit limb form
571 temp32 = prime->val[0];
572 for (i = 0; i < 8; i++) {
573 temp32 |= prime->val[i + 1] << (30-2*i);
574 us.a[i] = pp[i] = temp32;
575 temp32 = prime->val[i + 1] >> (2+2*i);
576 }
577 temp32 = x->val[0];
578 for (i = 0; i < 8; i++) {
579 temp32 |= x->val[i + 1] << (30-2*i);
580 vr.a[i] = temp32;
581 temp32 = x->val[i + 1] >> (2+2*i);
582 }
583 us.len1 = 8;
584 vr.len1 = 8;
585 // set s = 1 and r = 0
586 us.a[8] = 1;
587 vr.a[8] = 0;
588 // set k = 0.
589 k = 0;
590
591 // only one of the numbers u,v can be even at any time. We
592 // let even point to that number and odd to the other.
593 // Initially the prime u is guaranteed to be odd.
594 odd = &us;
595 even = &vr;
596
597 // u = prime, v = x
598 // r = 0 , s = 1
599 // k = 0
600 for (;;) {
601 // invariants:
602 // let u = limbs us.a[0..u.len1-1] in little endian,
603 // let s = limbs us.a[u.len..8] in big endian,
604 // let v = limbs vr.a[0..u.len1-1] in little endian,
605 // let r = limbs vr.a[u.len..8] in big endian,
606 // r,s >= 0 ; u,v >= 1
607 // x*-r = u*2^k mod prime
608 // x*s = v*2^k mod prime
609 // u*s + v*r = prime
610 // floor(log2(u)) + floor(log2(v)) + k <= 510
611 // max(u,v) <= 2^k (*) see comment at end of loop
612 // gcd(u,v) = 1
613 // {odd,even} = {&us, &vr}
614 // odd->a[0] and odd->a[8] are odd
615 // even->a[0] or even->a[8] is even
616 //
617 // first u/v are large and r/s small
618 // later u/v are small and r/s large
619 assert(odd->a[0] & 1);
620 assert(odd->a[8] & 1);
621
622 // adjust length of even.
623 while (even->a[even->len1 - 1] == 0) {
624 even->len1--;
625 // if input was 0, return.
626 // This simple check prevents crashing with stack underflow
627 // or worse undesired behaviour for illegal input.
628 if (even->len1 < 0)
629 return;
630 }
631
632 // reduce even->a while it is even
633 while (even->a[0] == 0) {
634 // shift right first part of even by a limb
635 // and shift left second part of even by a limb.
636 for (i = 0; i < 8; i++) {
637 even->a[i] = even->a[i+1];
638 }
639 even->a[i] = 0;
640 even->len1--;
641 k += 32;
642 }
643 // count up to 32 zero bits of even->a.
644 j = 0;
645 while ((even->a[0] & (1 << j)) == 0) {
646 j++;
647 }
648 if (j > 0) {
649 // shift first part of even right by j bits.
650 for (i = 0; i + 1 < even->len1; i++) {
651 even->a[i] = (even->a[i] >> j) | (even->a[i + 1] << (32 - j));
652 }
653 even->a[i] = (even->a[i] >> j);
654 if (even->a[i] == 0) {
655 even->len1--;
656 } else {
657 i++;
658 }
659
660 // shift second part of even left by j bits.
661 for (; i < 8; i++) {
662 even->a[i] = (even->a[i] << j) | (even->a[i + 1] >> (32 - j));
663 }
664 even->a[i] = (even->a[i] << j);
665 // add j bits to k.
666 k += j;
667 }
668 // invariant is reestablished.
669 // now both a[0] are odd.
670 assert(odd->a[0] & 1);
671 assert(odd->a[8] & 1);
672 assert(even->a[0] & 1);
673 assert((even->a[8] & 1) == 0);
674
675 // cmp > 0 if us.a[0..len1-1] > vr.a[0..len1-1],
676 // cmp = 0 if equal, < 0 if less.
677 cmp = us.len1 - vr.len1;
678 if (cmp == 0) {
679 i = us.len1 - 1;
680 while (i >= 0 && us.a[i] == vr.a[i]) i--;
681 // both are equal to 1 and we are done.
682 if (i == -1)
683 break;
684 cmp = us.a[i] > vr.a[i] ? 1 : -1;
685 }
686 if (cmp > 0) {
687 even = &us;
688 odd = &vr;
689 } else {
690 even = &vr;
691 odd = &us;
692 }
693
694 // now even > odd.
695
696 // even->a[0..len1-1] = (even->a[0..len1-1] - odd->a[0..len1-1]);
697 temp = 1;
698 for (i = 0; i < odd->len1; i++) {
699 temp += 0xFFFFFFFFull + even->a[i] - odd->a[i];
700 even->a[i] = temp & 0xFFFFFFFF;
701 temp >>= 32;
702 }
703 for (; i < even->len1; i++) {
704 temp += 0xFFFFFFFFull + even->a[i];
705 even->a[i] = temp & 0xFFFFFFFF;
706 temp >>= 32;
707 }
708 // odd->a[len1..8] = (odd->b[len1..8] + even->b[len1..8]);
709 temp = 0;
710 for (i = 8; i >= even->len1; i--) {
711 temp += (uint64_t) odd->a[i] + even->a[i];
712 odd->a[i] = temp & 0xFFFFFFFF;
713 temp >>= 32;
714 }
715 for (; i >= odd->len1; i--) {
716 temp += (uint64_t) odd->a[i];
717 odd->a[i] = temp & 0xFFFFFFFF;
718 temp >>= 32;
719 }
720 // note that
721 // if u > v:
722 // u'2^k = (u - v) 2^k = x(-r) - xs = x(-(r+s)) = x(-r') mod prime
723 // u's' + v'r' = (u-v)s + v(r+s) = us + vr
724 // if u < v:
725 // v'2^k = (v - u) 2^k = xs - x(-r) = x(s+r) = xs' mod prime
726 // u's' + v'r' = u(s+r) + (v-u)r = us + vr
727
728 // even->a[0] is difference between two odd numbers, hence even.
729 // odd->a[8] is sum of even and odd number, hence odd.
730 assert(odd->a[0] & 1);
731 assert(odd->a[8] & 1);
732 assert((even->a[0] & 1) == 0);
733
734 // The invariants are (almost) reestablished.
735 // The invariant max(u,v) <= 2^k can be invalidated at this point,
736 // because odd->a[len1..8] was changed. We only have
737 //
738 // odd->a[len1..8] <= 2^{k+1}
739 //
740 // Since even->a[0] is even, k will be incremented at the beginning
741 // of the next loop while odd->a[len1..8] remains unchanged.
742 // So after that, odd->a[len1..8] <= 2^k will hold again.
743 }
744 // In the last iteration we had u = v and gcd(u,v) = 1.
745 // Hence, u=1, v=1, s+r = prime, k <= 510, 2^k > max(s,r) >= prime/2
746 // This implies 0 <= s < prime and 255 <= k <= 510.
747 //
748 // The invariants also give us x*s = 2^k mod prime,
749 // hence s = 2^k * x^-1 mod prime.
750 // We need to compute s/2^k mod prime.
751
752 // First we compute inverse = -prime^-1 mod 2^32, which we need later.
753 // We use the Explicit Quadratic Modular inverse algorithm.
754 // http://arxiv.org/pdf/1209.6626.pdf
755 // a^-1 = (2-a) * PROD_i (1 + (a - 1)^(2^i)) mod 2^32
756 // the product will converge quickly, because (a-1)^(2^i) will be
757 // zero mod 2^32 after at most five iterations.
758 // We want to compute -prime^-1 so we start with (pp[0]-2).
759 assert(pp[0] & 1);
760 uint32_t amone = pp[0]-1;
761 uint32_t inverse = pp[0] - 2;
762 while (amone) {
763 amone *= amone;
764 inverse *= (amone + 1);
765 }
766
767 while (k >= 32) {
768 // compute s / 2^32 modulo prime.
769 // Idea: compute factor, such that
770 // s + factor*prime mod 2^32 == 0
771 // i.e. factor = s * -1/prime mod 2^32.
772 // Then compute s + factor*prime and shift right by 32 bits.
773 uint32_t factor = (inverse * us.a[8]) & 0xffffffff;
774 temp = us.a[8] + (uint64_t) pp[0] * factor;
775 assert((temp & 0xffffffff) == 0);
776 temp >>= 32;
777 for (i = 0; i < 7; i++) {
778 temp += us.a[8-(i+1)] + (uint64_t) pp[i+1] * factor;
779 us.a[8-i] = temp & 0xffffffff;
780 temp >>= 32;
781 }
782 us.a[8-i] = temp & 0xffffffff;
783 k -= 32;
784 }
785 if (k > 0) {
786 // compute s / 2^k modulo prime.
787 // Same idea: compute factor, such that
788 // s + factor*prime mod 2^k == 0
789 // i.e. factor = s * -1/prime mod 2^k.
790 // Then compute s + factor*prime and shift right by k bits.
791 uint32_t mask = (1 << k) - 1;
792 uint32_t factor = (inverse * us.a[8]) & mask;
793 temp = (us.a[8] + (uint64_t) pp[0] * factor) >> k;
794 assert(((us.a[8] + pp[0] * factor) & mask) == 0);
795 for (i = 0; i < 7; i++) {
796 temp += (us.a[8-(i+1)] + (uint64_t) pp[i+1] * factor) << (32 - k);
797 us.a[8-i] = temp & 0xffffffff;
798 temp >>= 32;
799 }
800 us.a[8-i] = temp & 0xffffffff;
801 }
802
803 // convert s to bignum style
804 temp32 = 0;
805 for (i = 0; i < 8; i++) {
806 x->val[i] = ((us.a[8-i] << (2 * i)) & 0x3FFFFFFFu) | temp32;
807 temp32 = us.a[8-i] >> (30 - 2 * i);
808 }
809 x->val[i] = temp32;
810
811 // let's wipe all temp buffers
812 MEMSET_BZERO(pp, sizeof(pp));
813 MEMSET_BZERO(&us, sizeof(us));
814 MEMSET_BZERO(&vr, sizeof(vr));
815}

888// a / 58 = a (+r)
889void bn_divmod58(bignum256 *a, uint32_t *r)
890{
891 int i;
892 uint32_t rem, tmp;
893 rem = a->val[8] % 58;
894 a->val[8] /= 58;
895 for (i = 7; i >= 0; i--) {
896 // invariants:
897 // rem = old(a) >> 30(i+1) % 58
898 // a[i+1..8] = old(a[i+1..8])/58
899 // a[0..i] = old(a[0..i])
900 // 2^30 == 18512790*58 + 4
901 tmp = rem * 4 + a->val[i];
902 // set a[i] = (rem * 2^30 + a[i])/58
903 // = rem * 18512790 + (rem * 4 + a[i])/58
904 a->val[i] = rem * 18512790 + (tmp / 58);
905 // set rem = (rem * 2^30 + a[i]) mod 58
906 // = (rem * 4 + a[i]) mod 58
907 rem = tmp % 58;
908 }
909 *r = rem;
910}
