43/* Initializing a sha3 context for given number of output bits */
44static void keccak_Init(SHA3_CTX *ctx, unsigned bits)
45{
46 /* NB: The Keccak capacity parameter = bits * 2 */
47 unsigned rate = 1600 - bits * 2;
48
49 memset(ctx, 0, sizeof(SHA3_CTX));
50 ctx->block_size = rate / 8;
51 assert(rate <= 1600 && (rate % 64) == 0);
52}

94/* Keccak theta() transformation */
95static void keccak_theta(uint64_t *A)
96{
97 unsigned int x;
98 uint64_t C[5], D[5];
99
100 for (x = 0; x < 5; x++) {
101 C[x] = A[x] ^ A[x + 5] ^ A[x + 10] ^ A[x + 15] ^ A[x + 20];
102 }
103 D[0] = ROTL64(C[1], 1) ^ C[4];
104 D[1] = ROTL64(C[2], 1) ^ C[0];
105 D[2] = ROTL64(C[3], 1) ^ C[1];
106 D[3] = ROTL64(C[4], 1) ^ C[2];
107 D[4] = ROTL64(C[0], 1) ^ C[3];
108
109 for (x = 0; x < 5; x++) {
110 A[x] ^= D[x];
111 A[x + 5] ^= D[x];
112 A[x + 10] ^= D[x];
113 A[x + 15] ^= D[x];
114 A[x + 20] ^= D[x];
115 }
116}

118/* Keccak pi() transformation */
119static void keccak_pi(uint64_t *A)
120{
121 uint64_t A1;
122 A1 = A[1];
123 A[ 1] = A[ 6];
124 A[ 6] = A[ 9];
125 A[ 9] = A[22];
126 A[22] = A[14];
127 A[14] = A[20];
128 A[20] = A[ 2];
129 A[ 2] = A[12];
130 A[12] = A[13];
131 A[13] = A[19];
132 A[19] = A[23];
133 A[23] = A[15];
134 A[15] = A[ 4];
135 A[ 4] = A[24];
136 A[24] = A[21];
137 A[21] = A[ 8];
138 A[ 8] = A[16];
139 A[16] = A[ 5];
140 A[ 5] = A[ 3];
141 A[ 3] = A[18];
142 A[18] = A[17];
143 A[17] = A[11];
144 A[11] = A[ 7];
145 A[ 7] = A[10];
146 A[10] = A1;
147 /* note: A[ 0] is left as is */
148}

150/* Keccak chi() transformation */
151static void keccak_chi(uint64_t *A)
152{
153 int i;
154 for (i = 0; i < 25; i += 5) {
155 uint64_t A0 = A[0 + i], A1 = A[1 + i];
156 A[0 + i] ^= ~A1 & A[2 + i];
157 A[1 + i] ^= ~A[2 + i] & A[3 + i];
158 A[2 + i] ^= ~A[3 + i] & A[4 + i];
159 A[3 + i] ^= ~A[4 + i] & A0;
160 A[4 + i] ^= ~A0 & A1;
161 }
162}

164static void sha3_permutation(uint64_t *state)
165{
166 int round;
167 for (round = 0; round < NumberOfRounds; round++)
168 {
169 keccak_theta(state);
170
171 /* apply Keccak rho() transformation */
172 state[ 1] = ROTL64(state[ 1], 1);
173 state[ 2] = ROTL64(state[ 2], 62);
174 state[ 3] = ROTL64(state[ 3], 28);
175 state[ 4] = ROTL64(state[ 4], 27);
176 state[ 5] = ROTL64(state[ 5], 36);
177 state[ 6] = ROTL64(state[ 6], 44);
178 state[ 7] = ROTL64(state[ 7], 6);
179 state[ 8] = ROTL64(state[ 8], 55);
180 state[ 9] = ROTL64(state[ 9], 20);
181 state[10] = ROTL64(state[10], 3);
182 state[11] = ROTL64(state[11], 10);
183 state[12] = ROTL64(state[12], 43);
184 state[13] = ROTL64(state[13], 25);
185 state[14] = ROTL64(state[14], 39);
186 state[15] = ROTL64(state[15], 41);
187 state[16] = ROTL64(state[16], 45);
188 state[17] = ROTL64(state[17], 15);
189 state[18] = ROTL64(state[18], 21);
190 state[19] = ROTL64(state[19], 8);

202 }

212static void sha3_process_block(uint64_t hash[25], const uint64_t *block, size_t block_size)
213{
214 /* expanded loop */
215 hash[ 0] ^= le2me_64(block[ 0]);
216 hash[ 1] ^= le2me_64(block[ 1]);
217 hash[ 2] ^= le2me_64(block[ 2]);
218 hash[ 3] ^= le2me_64(block[ 3]);
219 hash[ 4] ^= le2me_64(block[ 4]);
220 hash[ 5] ^= le2me_64(block[ 5]);
221 hash[ 6] ^= le2me_64(block[ 6]);
222 hash[ 7] ^= le2me_64(block[ 7]);
223 hash[ 8] ^= le2me_64(block[ 8]);
224 /* if not sha3-512 */
225 if (block_size > 72) {
226 hash[ 9] ^= le2me_64(block[ 9]);
227 hash[10] ^= le2me_64(block[10]);
228 hash[11] ^= le2me_64(block[11]);
229 hash[12] ^= le2me_64(block[12]);
230 /* if not sha3-384 */
231 if (block_size > 104) {
232 hash[13] ^= le2me_64(block[13]);
233 hash[14] ^= le2me_64(block[14]);
234 hash[15] ^= le2me_64(block[15]);
235 hash[16] ^= le2me_64(block[16]);
236 /* if not sha3-256 */
237 if (block_size > 136) {
238 hash[17] ^= le2me_64(block[17]);
239#ifdef FULL_SHA3_FAMILY_SUPPORT
240 /* if not sha3-224 */

252 }
253 }

268void sha3_Update(SHA3_CTX *ctx, const unsigned char *msg, size_t size)
269{
270 size_t idx = (size_t)ctx->rest;
271 size_t block_size = (size_t)ctx->block_size;
272
273 if (ctx->rest & SHA3_FINALIZED) return; /* too late for additional input */
274 ctx->rest = (unsigned)((ctx->rest + size) % block_size);
275
276 /* fill partial block */
277 if (idx) {
278 size_t left = block_size - idx;
279 memcpy((char*)ctx->message + idx, msg, (size < left ? size : left));
280 if (size < left) return;
281
282 /* process partial block */
283 sha3_process_block(ctx->hash, ctx->message, block_size);
284 msg += left;
285 size -= left;
286 }
287 while (size >= block_size) {
288 uint64_t* aligned_message_block;
289 if (IS_ALIGNED_64(msg)) {
290 /* the most common case is processing of an already aligned message
291 without copying it */
292 aligned_message_block = (uint64_t*)(void*)msg;
293 } else {
294 memcpy(ctx->message, msg, block_size);
295 aligned_message_block = ctx->message;
296 }
297
298 sha3_process_block(ctx->hash, aligned_message_block, block_size);
299 msg += block_size;
300 size -= block_size;
301 }
302 if (size) {
303 memcpy(ctx->message, msg, size); /* save leftovers */
304 }
305}

313void sha3_Final(SHA3_CTX *ctx, unsigned char* result)
314{
315 size_t digest_length = 100 - ctx->block_size / 2;
316 const size_t block_size = ctx->block_size;
317
318 if (!(ctx->rest & SHA3_FINALIZED))
319 {
320 /* clear the rest of the data queue */
321 memset((char*)ctx->message + ctx->rest, 0, block_size - ctx->rest);
322 ((char*)ctx->message)[ctx->rest] |= 0x06;
323 ((char*)ctx->message)[block_size - 1] |= 0x80;
324
325 /* process final block */
326 sha3_process_block(ctx->hash, ctx->message, block_size);
327 ctx->rest = SHA3_FINALIZED; /* mark context as finalized */
328 }
329
330 assert(block_size > digest_length);
331 if (result) me64_to_le_str(result, ctx->hash, digest_length);
332}
