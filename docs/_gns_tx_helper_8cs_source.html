25 public static CalldataBytes CalculateCalldataBytesZeroNonzero(string calldata)
26 {
27 var calldataBuf = Encoding.UTF8.GetBytes(calldata.Replace("0x", ""));
28
29 int calldataZeroBytes = calldataBuf.Count(ch => ch == 0);
30 int calldataNonzeroBytes = calldataBuf.Length - calldataZeroBytes;
31
32 return new CalldataBytes(calldataZeroBytes, calldataNonzeroBytes);
33 }

35 public static int CalculateCalldataCost(string msgData, int gtxDataNonZero, int gtxDataZero)
36 {
37 var calldataBytesZeroNonzero = CalculateCalldataBytesZeroNonzero(msgData);
38 return calldataBytesZeroNonzero.CalldataZeroBytes * gtxDataZero + calldataBytesZeroNonzero.CalldataNonzeroBytes * gtxDataNonZero;
39 }

178 public static string GetRelayRequestID(Dictionary<string, object> relayRequest, string signature)
179 {
180 var parameters = new ABIValue[]
181 {
182 new("address", ((Dictionary<string, object>)relayRequest["request"])["from"]),
183 new("uint256", ((Dictionary<string, object>)relayRequest["request"])["nonce"]),
184 new("bytes", signature.HexToByteArray()),
185 };
186
187 var hash = Sha3Keccack.Current.CalculateHash(new ABIEncode().GetABIEncoded(parameters));
188
189 var rawRelayRequestId = hash.ToHex().PadLeft(64, '0');
190 const int prefixSize = 8;
191 var prefixedRelayRequestId = new string('0', prefixSize) + rawRelayRequestId[prefixSize..];
192
193 return $"0x{prefixedRelayRequestId}";
194 }

228 public static async Task<string> HandleGsnResponse(HttpResponseMessage response, Web3 ethClient)
229 {
230 var responseContent = await response.Content.ReadAsStringAsync();
231 var responseMap = JsonConvert.DeserializeObject<Dictionary<string, object>>(responseContent) ?? new Dictionary<string, object>();
232
233 if (responseMap.ContainsKey("error"))
234 {
235 throw new Exception($"RelayError: {responseMap["error"]}");
236 }
237
238 var txHash = $"0x{Sha3Keccack.Current.CalculateHashFromHex(responseMap["signedTx"].ToString())}";
239 TransactionReceipt receipt;
240 do
241 {
242 receipt = await ethClient.Eth.Transactions.GetTransactionReceipt.SendRequestAsync(txHash);
243 await Task.Delay(2000);
244 }
245 while (receipt == null);
246
247 return txHash;
248 }

262 public static BigInteger ParseUnits(string value, int decimals)
263 {
264 var baseValue = BigInteger.Pow(10, decimals);
265 var parts = value.Split('.');
266 var wholePart = BigInteger.Parse(parts[0]);
267 var fractionalPart = parts.Length > 1 ? BigInteger.Parse(parts[1].PadRight(decimals, '0')) : BigInteger.Zero;
268
269 return wholePart * baseValue + fractionalPart;
270 }
