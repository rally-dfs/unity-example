132int data_from_mnemonic(const char *mnemonic, uint8_t *data) {
133 if (!mnemonic) {
134 return 0;
135 }
136
137 uint32_t i, n;
138
139 i = 0; n = 0;
140 while (mnemonic[i]) {
141 if (mnemonic[i] == ' ') {
142 n++;
143 }
144 i++;
145 }
146 n++;
147 // check number of words
148 if (n != 12 && n != 18 && n != 24) {
149 return 0;
150 }
151
152 char current_word[10];
153 uint32_t j, k, ki, bi;
154 //uint8_t bits[32 + 1];
155 memset(data, 0, MAXIMUM_BIP39_DATA_LENGTH);
156 i = 0; bi = 0;
157 while (mnemonic[i]) {
158 j = 0;
159 while (mnemonic[i] != ' ' && mnemonic[i] != 0) {
160 if (j >= sizeof(current_word) - 1) {
161 return 0;
162 }
163 current_word[j] = mnemonic[i];
164 i++; j++;
165 }
166 current_word[j] = 0;
167 if (mnemonic[i] != 0) i++;
168 k = 0;
169 for (;;) {
170 if (!wordlist[k]) { // word not found
171 return 0;
172 }
173 if (strcmp(current_word, wordlist[k]) == 0) { // word found on index k
174 for (ki = 0; ki < 11; ki++) {
175 if (k & (1 << (10 - ki))) {
176 data[bi / 8] |= 1 << (7 - (bi % 8));
177 }
178 bi++;
179 }
180 break;
181 }
182 k++;
183 }
184 }
185
186 if (bi != n * 11) {
187 return 0;
188 }
189
190 char expectedChecksum = data[n * 4 / 3];
191
192 // Compute the checksum
193 uint8_t computedChecksum[SHA256_DIGEST_LENGTH];
194 sha256_Raw(data, n * 4 / 3, computedChecksum);
195
196 // Compute checksum mask
197 int mask = 0;
198 for (int i = 0; i < (n / 3); i++) {
199 mask |= (0x80 >> i);
200 }
201
202 // Check checksum bits
203 if ((computedChecksum[0] & mask) != (expectedChecksum & mask)) {
204 return 0;
205 }
206
207 return n * 4 / 3;

221{
222 int passphraselen = (int)strlen(passphrase);
223#if USE_BIP39_CACHE
224 int mnemoniclen = (int)strlen(mnemonic);
225 // check cache
226 if (mnemoniclen < 256 && passphraselen < 64) {
227 for (int i = 0; i < BIP39_CACHE_SIZE; i++) {
228 if (!bip39_cache[i].set) continue;
229 if (strcmp(bip39_cache[i].mnemonic, mnemonic) != 0) continue;
230 if (strcmp(bip39_cache[i].passphrase, passphrase) != 0) continue;
231 // found the correct entry
232 memcpy(seed, bip39_cache[i].seed, 512 / 8);
233 return;
234 }
235 }
236#endif
237 uint8_t salt[8 + 256];
238 memcpy(salt, "mnemonic", 8);
239 memcpy(salt + 8, passphrase, passphraselen);
240 PBKDF2_HMAC_SHA512_CTX pctx;
241 pbkdf2_hmac_sha512_Init(&pctx, (const uint8_t *)mnemonic, (int)strlen(mnemonic), salt, passphraselen + 8);
242 if (progress_callback) {
243 progress_callback(0, BIP39_PBKDF2_ROUNDS);
244 }
245 for (int i = 0; i < 16; i++) {
246 pbkdf2_hmac_sha512_Update(&pctx, BIP39_PBKDF2_ROUNDS / 16);
247 if (progress_callback) {
248 progress_callback((i + 1) * BIP39_PBKDF2_ROUNDS / 16, BIP39_PBKDF2_ROUNDS);
249 }
250 }
251 pbkdf2_hmac_sha512_Final(&pctx, seed);
252#if USE_BIP39_CACHE

254 if (mnemoniclen < 256 && passphraselen < 64) {
255 bip39_cache[bip39_cache_index].set = true;
256 strcpy(bip39_cache[bip39_cache_index].mnemonic, mnemonic);

258 memcpy(bip39_cache[bip39_cache_index].seed, seed, 512 / 8);
259 bip39_cache_index = (bip39_cache_index + 1) % BIP39_CACHE_SIZE;
260 }
