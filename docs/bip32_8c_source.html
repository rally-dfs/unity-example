107int hdnode_from_seed(const uint8_t *seed, int seed_len, const char* curve, HDNode *out)
108{
109 uint8_t I[32 + 32];
110 memset(out, 0, sizeof(HDNode));
111 out->depth = 0;
112 out->child_num = 0;
113 out->curve = get_curve_by_name(curve);
114 if (out->curve == 0) {
115 return 0;
116 }
117 hmac_sha512((const uint8_t*) out->curve->bip32_name,
118 (uint32_t)strlen(out->curve->bip32_name), seed, seed_len, I);
119
120 if (out->curve->params) {
121 bignum256 a;
122 while (true) {
123 bn_read_be(I, &a);
124 if (!bn_is_zero(&a) // != 0
125 && bn_is_less(&a, &out->curve->params->order)) { // < order
126 break;
127 }
128 hmac_sha512((const uint8_t*) out->curve->bip32_name,
129 (uint32_t)strlen(out->curve->bip32_name), I, sizeof(I), I);
130 }
131 MEMSET_BZERO(&a, sizeof(a));
132 }
133 memcpy(out->private_key, I, 32);
134 memcpy(out->chain_code, I + 32, 32);
135 MEMSET_BZERO(out->public_key, sizeof(out->public_key));
136 MEMSET_BZERO(I, sizeof(I));
137 return 1;
138}

140uint32_t hdnode_fingerprint(HDNode *node)
141{
142 uint8_t digest[32];
143 uint32_t fingerprint;
144
145 hdnode_fill_public_key(node);
146 sha256_Raw(node->public_key, 33, digest);
147 ripemd160(digest, 32, digest);
148 fingerprint = (digest[0] << 24) + (digest[1] << 16) + (digest[2] << 8) + digest[3];
149 MEMSET_BZERO(digest, sizeof(digest));
150 return fingerprint;

153int hdnode_private_ckd(HDNode *inout, uint32_t i)
154{
155 uint8_t data[1 + 32 + 4];
156 uint8_t I[32 + 32];
157 bignum256 a, b;
158
159 if (i & 0x80000000) { // private derivation
160 data[0] = 0;
161 memcpy(data + 1, inout->private_key, 32);
162 } else { // public derivation
163 if (!inout->curve->params) {
164 return 0;
165 }
166 hdnode_fill_public_key(inout);
167 memcpy(data, inout->public_key, 33);
168 }
169 write_be(data + 33, i);
170
171 bn_read_be(inout->private_key, &a);
172
173 hmac_sha512(inout->chain_code, 32, data, sizeof(data), I);
174 if (inout->curve->params) {
175 while (true) {
176 bool failed = false;
177 bn_read_be(I, &b);
178 if (!bn_is_less(&b, &inout->curve->params->order)) { // >= order
179 failed = true;
180 } else {
181 bn_addmod(&b, &a, &inout->curve->params->order);
182 bn_mod(&b, &inout->curve->params->order);
183 if (bn_is_zero(&b)) {
184 failed = true;
185 }
186 }
187
188 if (!failed) {
189 bn_write_be(&b, inout->private_key);
190 break;
191 }
192
193 data[0] = 1;
194 memcpy(data + 1, I + 32, 32);
195 hmac_sha512(inout->chain_code, 32, data, sizeof(data), I);
196 }
197 } else {
198 memcpy(inout->private_key, I, 32);
199 }
200
201 memcpy(inout->chain_code, I + 32, 32);
202 inout->depth++;
203 inout->child_num = i;
204 MEMSET_BZERO(inout->public_key, sizeof(inout->public_key));
205
206 // making sure to wipe our memory
207 MEMSET_BZERO(&a, sizeof(a));
208 MEMSET_BZERO(&b, sizeof(b));
209 MEMSET_BZERO(I, sizeof(I));
210 MEMSET_BZERO(data, sizeof(data));
211 return 1;
212}

214int hdnode_public_ckd(HDNode *inout, uint32_t i)
215{
216 uint8_t data[1 + 32 + 4];
217 uint8_t I[32 + 32];
218 curve_point a, b;
219 bignum256 c;
220
221 if (i & 0x80000000) { // private derivation
222 return 0;
223 } else { // public derivation
224 if (!inout->curve->params) {
225 return 0;
226 }
227 memcpy(data, inout->public_key, 33);
228 }
229 write_be(data + 33, i);
230 memset(inout->private_key, 0, 32);
231
232 if (!ecdsa_read_pubkey(inout->curve->params, inout->public_key, &a)) {
233 return 0;
234 }
235
236 while (true) {
237 bool failed = false;
238 hmac_sha512(inout->chain_code, 32, data, sizeof(data), I);
239 bn_read_be(I, &c);
240 if (!bn_is_less(&c, &inout->curve->params->order)) { // >= order
241 failed = true;
242 } else {
243 scalar_multiply(inout->curve->params, &c, &b); // b = c * G
244 point_add(inout->curve->params, &a, &b); // b = a + b
245 if (point_is_infinity(&b)) {
246 failed = true;
247 }
248 }
249
250 if (!failed) {
251 inout->public_key[0] = 0x02 | (b.y.val[0] & 0x01);
252 bn_write_be(&b.x, inout->public_key + 1);
253 break;
254 }
255
256 data[0] = 1;
257 memcpy(data + 1, I + 32, 32);
258 }
259
260 inout->depth++;
261 inout->child_num = i;
262 memcpy(inout->chain_code, I + 32, 32);
263
264 // Wipe all stack data.
265 MEMSET_BZERO(data, sizeof(data));
266 MEMSET_BZERO(I, sizeof(I));
267 MEMSET_BZERO(&a, sizeof(a));
268 MEMSET_BZERO(&b, sizeof(b));
269 MEMSET_BZERO(&c, sizeof(c));
270
271 return 1;
272}

326int hdnode_private_ckd_cached(HDNode *inout, const uint32_t *i, size_t i_count)
327{
328 if (i_count == 0) {
329 return 1;
330 }
331 if (i_count == 1) {
332 if (hdnode_private_ckd(inout, i[0]) == 0) return 0;
333 return 1;
334 }
335
336 bool found = false;
337 // if root is not set or not the same
338 if (!private_ckd_cache_root_set || memcmp(&private_ckd_cache_root, inout, sizeof(HDNode)) != 0) {
339 // clear the cache
340 private_ckd_cache_index = 0;
341 memset(private_ckd_cache, 0, sizeof(private_ckd_cache));
342 // setup new root
343 memcpy(&private_ckd_cache_root, inout, sizeof(HDNode));
344 private_ckd_cache_root_set = true;
345 } else {
346 // try to find parent
347 int j;
348 for (j = 0; j < BIP32_CACHE_SIZE; j++) {
349 if (private_ckd_cache[j].set &&
350 private_ckd_cache[j].depth == i_count - 1 &&
351 memcmp(private_ckd_cache[j].i, i, (i_count - 1) * sizeof(uint32_t)) == 0 &&
352 private_ckd_cache[j].node.curve == inout->curve) {
353 memcpy(inout, &(private_ckd_cache[j].node), sizeof(HDNode));
354 found = true;
355 break;
356 }
357 }
358 }
359
360 // else derive parent
361 if (!found) {
362 size_t k;
363 for (k = 0; k < i_count - 1; k++) {
364 if (hdnode_private_ckd(inout, i[k]) == 0) return 0;
365 }
366 // and save it
367 memset(&(private_ckd_cache[private_ckd_cache_index]), 0, sizeof(private_ckd_cache[private_ckd_cache_index]));
368 private_ckd_cache[private_ckd_cache_index].set = true;
369 private_ckd_cache[private_ckd_cache_index].depth = i_count - 1;
370 memcpy(private_ckd_cache[private_ckd_cache_index].i, i, (i_count - 1) * sizeof(uint32_t));
371 memcpy(&(private_ckd_cache[private_ckd_cache_index].node), inout, sizeof(HDNode));
372 private_ckd_cache_index = (private_ckd_cache_index + 1) % BIP32_CACHE_SIZE;
373 }
374
375 if (hdnode_private_ckd(inout, i[i_count - 1]) == 0) return 0;
376

523int hdnode_deserialize(const char *str, HDNode *node, uint32_t *fingerprint)
524{
525 uint8_t node_data[78];

531 uint32_t version = read_be(node_data);

533 MEMSET_BZERO(node->private_key, sizeof(node->private_key));
534 memcpy(node->public_key, node_data + 45, 33);
535 } else if (version == 0x0488ADE4) { // private node
536 if (node_data[45]) { // invalid data
537 return -2;
538 }
539 memcpy(node->private_key, node_data + 46, 32);
540 MEMSET_BZERO(node->public_key, sizeof(node->public_key));
541 } else {
542 return -3; // invalid version
543 }
544 node->depth = node_data[4];
545 if (fingerprint) {
546 *fingerprint = read_be(node_data + 5);
547 }
548 node->child_num = read_be(node_data + 9);
549 memcpy(node->chain_code, node_data + 13, 32);
